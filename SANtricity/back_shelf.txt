      // Put the encoded password in local storage and remove the sid param from the URL.
      if (window.location.search.indexOf('sid=') >= 0) {
         // Extract the sid parameter from the URL.
         var regex = new RegExp('[\\?&]sid=([^&]+)($|&)');
         // Capture sid value.
         var sidParam = regex.exec(window.location.href);
         try {
            sessionStorage.setItem('sid', sidParam[1]);
            //Remove the sid param from the search string.
            var newHref = window.location.href.replace(/([\\?&])sid=[^&]+($|&)/,
                  function(searchValue, replaceValue1, replaceValue2) {
                     // sid is not the first param. Leave in the sign following it, if there is one.
                     if (replaceValue1 === '&') {
                        return replaceValue2;
                     } else {
                        // sid is first param. If it is followed by a param, keep the '?', otherwise
                        // remove everything.
                        return replaceValue2 === '&' ? '?' : '';
                     }
                  });
            window.history.replaceState({}, '', newHref);
         } catch (e) {
            console.log('Could not set the sid.');
         }
      }
   </script>
